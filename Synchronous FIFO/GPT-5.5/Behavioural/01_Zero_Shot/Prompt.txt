targeting Nexys A7 (xc7a100tcsg324-2) in BEHAVIORAL form. 
 
/* ═══ PARAMETERS ════════════════════════════════ */ 
localparam DATA_WIDTH = 16; 
localparam DEPTH      = 16; 
localparam ADDR_WIDTH = 4; 
localparam PTR_WIDTH  = ADDR_WIDTH + 1; // 5-bit: MSB=wrap 
 
/* ═══ MANDATORY BEHAVIORAL LOOPS ════════════════ */

//   end

// LOOP B — BRAM write port (inference)

// LOOP C — BRAM read port, read-first mode

//     dout_pre <= mem[rd_ptr[ADDR_WIDTH-1:0]];

// LOOP D — Flag logic (registered, no latches) 
//   always @(posedge clk) begin 
//     if (!rst_n) begin full<=0; empty<=1; ... end 
//     else begin

//       empty <= (wr_ptr_next == rd_ptr_next); 
//       // ... almost_full, almost_empty, count 
//     end 
//   end 
// 
// LOOP E — Output register (registered dout for timing closure) 
//   always @(posedge clk) 
//     if (!rst_n) dout <= {DATA_WIDTH{1'b0}}; 
//     else if (rd_en && !empty) dout <= dout_pre; 
// 
// LOOP F — Generate-for (parameterized reset of wide signals) 
//   genvar i; 
//   generate 
//     for (i=0; i<PTR_WIDTH; i=i+1) begin : ptr_reset_gen 
//       // structural hook for per-bit reset tie-off if needed 
//     end 
//   endgenerate 
// 
// LOOP G — For loop inside always (initialise count register) 
//   integer k; 
//   always @(posedge clk) 
//     if (!rst_n) for (k=0; k<DEPTH; k=k+1) count_arr[k] <= 1'b0;
